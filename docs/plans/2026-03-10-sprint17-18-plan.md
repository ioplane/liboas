**Public API:**

```c
#ifndef LIBOAS_PARSER_URI_H
#define LIBOAS_PARSER_URI_H

#include <stddef.h>
#include <stdint.h>

typedef struct {
    const char *scheme;       /* "http", "https", "file", nullptr for relative */
    const char *host;         /* "example.com", nullptr for relative/file */
    uint16_t port;            /* 0 = default (80/443) */
    const char *path;         /* "/schemas/pet.yaml" */
    const char *query;        /* after ?, nullptr if none */
    const char *fragment;     /* after #, nullptr if none (JSON Pointer) */
    bool is_absolute;         /* has scheme */
    bool is_fragment_only;    /* starts with # */
} oas_uri_t;

/**
 * @brief Parse URI string into components (RFC 3986).
 * @param uri     Input URI string (e.g., "https://host/path#/fragment")
 * @param arena   Arena for string allocations
 * @param out     Parsed URI components
 * @return 0 on success, negative errno on error
 */
[[nodiscard]] int oas_uri_parse(const char *uri, oas_arena_t *arena, oas_uri_t *out);

/**
 * @brief Resolve relative URI against base URI (RFC 3986 S5).
 * @param base    Base URI (from source document or $id)
 * @param ref     Reference URI to resolve
 * @param arena   Arena for result string
 * @param out     Resolved absolute URI string
 * @return 0 on success, negative errno on error
 */
[[nodiscard]] int oas_uri_resolve(const oas_uri_t *base, const char *ref,
                                   oas_arena_t *arena, char **out);

/**
 * @brief Check if path contains traversal sequences (../../).
 * @return true if path is safe, false if traversal detected
 */
bool oas_uri_path_is_safe(const char *path);

#endif /* LIBOAS_PARSER_URI_H */
```

**Implementation (`oas_uri.c`):**

```c
#include "oas_uri.h"

#include <liboas/oas_alloc.h>

#include <string.h>
#include <errno.h>

[[nodiscard]] int oas_uri_parse(const char *uri, oas_arena_t *arena, oas_uri_t *out)
{
    if (!uri || !arena || !out)
        return -EINVAL;

    memset(out, 0, sizeof(*out));
    const char *p = uri;

    /* Fragment-only: "#/components/schemas/Pet" */
    if (*p == '#') {
        out->is_fragment_only = true;
        out->fragment = oas_arena_strdup(arena, p + 1);
        return out->fragment ? 0 : -ENOMEM;
    }

    /* Check for scheme: letters followed by "://" */
    const char *colon = strchr(p, ':');
    if (colon && colon > p && colon[1] == '/' && colon[2] == '/') {
        size_t scheme_len = (size_t)(colon - p);
        char *s = oas_arena_alloc(arena, scheme_len + 1, 1);
        if (!s) return -ENOMEM;
        memcpy(s, p, scheme_len);
        s[scheme_len] = '\0';
        out->scheme = s;
        out->is_absolute = true;
        p = colon + 3; /* skip "://" */

        /* Parse host[:port] */
        const char *path_start = strchr(p, '/');
        const char *host_end = path_start ? path_start : p + strlen(p);

        /* Check for port */
        const char *port_sep = memchr(p, ':', (size_t)(host_end - p));
        if (port_sep) {
            size_t hlen = (size_t)(port_sep - p);
            char *h = oas_arena_alloc(arena, hlen + 1, 1);
            if (!h) return -ENOMEM;
            memcpy(h, p, hlen);
            h[hlen] = '\0';
            out->host = h;

            /* Parse port number */
            out->port = 0;
            for (const char *pp = port_sep + 1; pp < host_end; pp++) {
                if (*pp < '0' || *pp > '9') return -EINVAL;
                out->port = (uint16_t)(out->port * 10 + (*pp - '0'));
            }
        } else {
            size_t hlen = (size_t)(host_end - p);
            char *h = oas_arena_alloc(arena, hlen + 1, 1);
            if (!h) return -ENOMEM;
            memcpy(h, p, hlen);
            h[hlen] = '\0';
            out->host = h;
        }

        p = path_start ? path_start : "";
    }

    /* Parse path */
    const char *query_start = strchr(p, '?');
    const char *frag_start = strchr(p, '#');
    const char *path_end = query_start ? query_start : frag_start ? frag_start : p + strlen(p);

    if (path_end > p) {
        size_t plen = (size_t)(path_end - p);
        char *path = oas_arena_alloc(arena, plen + 1, 1);
        if (!path) return -ENOMEM;
        memcpy(path, p, plen);
        path[plen] = '\0';
        out->path = path;
    }

    /* Parse query */
    if (query_start) {
        const char *qend = frag_start ? frag_start : query_start + strlen(query_start);
        size_t qlen = (size_t)(qend - query_start - 1);
        char *q = oas_arena_alloc(arena, qlen + 1, 1);
        if (!q) return -ENOMEM;
        memcpy(q, query_start + 1, qlen);
        q[qlen] = '\0';
        out->query = q;
    }

    /* Parse fragment */
    if (frag_start) {
        out->fragment = oas_arena_strdup(arena, frag_start + 1);
        if (!out->fragment) return -ENOMEM;
    }

    return 0;
}

[[nodiscard]] int oas_uri_resolve(const oas_uri_t *base, const char *ref,
                                   oas_arena_t *arena, char **out)
{
    if (!base || !ref || !arena || !out)
        return -EINVAL;

    oas_uri_t ref_parsed;
    int rc = oas_uri_parse(ref, arena, &ref_parsed);
    if (rc < 0) return rc;

    /* Fragment-only: use base document */
    if (ref_parsed.is_fragment_only) {
        /* Reconstruct base URL + new fragment */
        size_t len = 0;
        if (base->scheme) len += strlen(base->scheme) + 3;
        if (base->host) len += strlen(base->host) + 8; /* :port */
        if (base->path) len += strlen(base->path);
        len += 1 + strlen(ref_parsed.fragment) + 1; /* #fragment\0 */

        char *result = oas_arena_alloc(arena, len, 1);
        if (!result) return -ENOMEM;

        char *p = result;
        if (base->scheme) {
            p += sprintf(p, "%s://", base->scheme);
            if (base->host) {
                p += sprintf(p, "%s", base->host);
                if (base->port) p += sprintf(p, ":%u", base->port);
            }
        }
        if (base->path) p += sprintf(p, "%s", base->path);
        (void)sprintf(p, "#%s", ref_parsed.fragment);
        *out = result;
        return 0;
    }

    /* Absolute URI: use as-is */
    if (ref_parsed.is_absolute) {
        *out = oas_arena_strdup(arena, ref);
        return *out ? 0 : -ENOMEM;
    }

    /* Relative path: resolve against base */
    if (!base->scheme || !base->host)
        return -EINVAL; /* can't resolve relative against relative */

    /* Merge base path directory + relative path */
    const char *base_dir = base->path;
    const char *last_slash = base_dir ? strrchr(base_dir, '/') : nullptr;
    size_t dir_len = last_slash ? (size_t)(last_slash - base_dir + 1) : 0;

    const char *ref_path = ref_parsed.path ? ref_parsed.path : "";
    size_t ref_len = strlen(ref_path);
    const char *ref_frag = ref_parsed.fragment;

    size_t total = strlen(base->scheme) + 3 + strlen(base->host) + 8 +
                   dir_len + ref_len + (ref_frag ? 1 + strlen(ref_frag) : 0) + 1;
    char *result = oas_arena_alloc(arena, total, 1);
    if (!result) return -ENOMEM;

    char *p = result;
    p += sprintf(p, "%s://%s", base->scheme, base->host);
    if (base->port) p += sprintf(p, ":%u", base->port);
    if (dir_len) { memcpy(p, base_dir, dir_len); p += dir_len; }
    memcpy(p, ref_path, ref_len); p += ref_len;
    if (ref_frag) p += sprintf(p, "#%s", ref_frag);
    *p = '\0';

    *out = result;
    return 0;
}

bool oas_uri_path_is_safe(const char *path)
{
    if (!path) return true;
    /* Reject path traversal */
    if (strstr(path, "/../") || strstr(path, "/..") == path + strlen(path) - 3)
        return false;
    if (strncmp(path, "../", 3) == 0 || strcmp(path, "..") == 0)
        return false;
    return true;
}
```

#define LIBOAS_PARSER_REF_CACHE_H

#include <stddef.h>

yyjson_val *oas_ref_cache_get(oas_ref_cache_t *cache, const char *uri);

 * @return 0 on success, -ENOMEM, -ENOSPC if cache full
 */

                                     yyjson_doc *doc, yyjson_val *root);

/**
 * @brief Get number of cached documents.
 */
size_t oas_ref_cache_count(const oas_ref_cache_t *cache);

/**
 * @brief Destroy cache and all owned yyjson_docs.
 */
void oas_ref_cache_destroy(oas_ref_cache_t *cache);

#endif /* LIBOAS_PARSER_REF_CACHE_H */
```

**Implementation:** Hash map (URI string → yyjson_doc + root). FNV-1a hash, open addressing. Owns yyjson_doc lifetime — `oas_ref_cache_destroy()` frees all docs.

 * @brief Fetch document via HTTP (built-in POSIX sockets).

 * @param max_redirects Max redirect hops (0 = default 5)

**Implementation (~150 LOC):**
- Parse URL → host, port (default 80), path
- `socket()` + `connect()` with `struct sockaddr_in`
- Send `GET /path HTTP/1.1\r\nHost: host\r\nConnection: close\r\n\r\n`
- `setsockopt(SO_RCVTIMEO)` for timeout
- Read response, parse status line, find Content-Length or read until close
- Follow 301/302/307/308 redirects (Location header), decrement counter
- Enforce max_size
- HTTPS URLs → return `-ENOTSUP` with message "HTTPS requires -DOAS_TLS=ON or custom fetch callback"

 * @brief Validate request and apply default values from schema.
 *
 * Creates a mutable copy of the request body JSON, applies default values
 * for missing fields, validates, and returns the modified body.
 *
 * @param doc       Compiled document
 * @param req       HTTP request (body will NOT be modified)
 * @param result    Validation result
 * @param arena     Arena for error allocation
 * @param out_body  Output: modified body JSON with defaults applied (caller frees with free())
 * @param out_len   Output: modified body length
 * @return 0 on success, negative errno on error
 */

    const oas_compiled_doc_t *doc,
    const oas_http_request_t *req,
    oas_validation_result_t *result,
    oas_arena_t *arena,
    char **out_body, size_t *out_len);
```

**Implementation (`oas_defaults.c`, ~150 LOC):**

```c
/* Recursively apply defaults from schema to mutable JSON doc */
static void apply_defaults(yyjson_mut_doc *mdoc, yyjson_mut_val *obj,
                            const oas_schema_t *schema)
{
    if (!schema || !obj || !yyjson_mut_is_obj(obj))
        return;

    /* For each property in schema */
    for (size_t i = 0; i < schema->properties_count; i++) {
        const char *name = schema->properties[i].name;
        const oas_schema_t *prop_schema = schema->properties[i].schema;

        yyjson_mut_val *existing = yyjson_mut_obj_get(obj, name);
        if (!existing && prop_schema->default_value) {
            /* Field missing, has default — apply it */
            yyjson_mut_val *key = yyjson_mut_strcpy(mdoc, name);
            yyjson_mut_val *val = yyjson_val_mut_copy(mdoc, prop_schema->default_value);
            if (key && val) {
                yyjson_mut_obj_add(obj, key, val);
            }
        } else if (existing && yyjson_mut_is_obj(existing) && prop_schema) {
            /* Recurse into nested objects */
            apply_defaults(mdoc, existing, prop_schema);
        }
    }
}
```

Workflow:
1. Find operation via path matching (reuse existing logic)
2. Parse request body into yyjson mutable doc
3. Find matching schema for the operation's request body media type
4. Recursively apply defaults
5. Serialize mutable doc back to JSON string
6. Validate the modified body
7. Return modified body in `out_body`
